library(RSSampling)
library(mnormt)
library(normalp)

mu<-c(10,8)
variance<-c(5,3)

a<-matrix(c(1,0.9,0.9,1),2,2)
v<-diag(variance)

Sigma<-v%*%a%*%v

x<-rmnorm(10000,mu,Sigma)
xx<-as.numeric(x[,1])
xy<-as.numeric(x[,2])
con.Rrss(xx,xy,m=4,r=2,type="tb",sets=TRUE,concomitant=FALSE,alpha=0.25)
